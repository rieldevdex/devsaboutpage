// 3D tilt on mousemove
const initTilt = (el) => {
  el.addEventListener('mousemove', e => {
    const r = el.getBoundingClientRect();
    const x = (e.clientX - r.left) / r.width  - .5;
    const y = (e.clientY - r.top)  / r.height - .5;
    el.style.transform =
      `perspective(800px)
       rotateY(${x*12}deg)
       rotateX(${-y*12}deg)`;
  });
};
